const grid = document.getElementById('grid');
const searchEl = document.getElementById('search');
const categoryEl = document.getElementById('category');

const emptyEl = document.getElementById('empty');
const coverCacheKey = 'bfml-cover-cache-v1';
let visibleBooks = [];
let observer;
let coverCache = {};
try { coverCache = JSON.parse(localStorage.getItem(coverCacheKey) || '{}'); } catch(e) { coverCache = {}; }

function saveCoverCache() {
  try { localStorage.setItem(coverCacheKey, JSON.stringify(coverCache)); } catch(e) {}
}

function normalise(text) { return (text || '').toString().toLowerCase().trim(); }
function coverKey(book) { return (book.title + '|' + book.author).toLowerCase(); }
function statusClass(status) {
  const s = normalise(status);
  if (s.includes('available')) return 'available';
  if (s.includes('in use')) return 'inuse';
  if (s.includes('just arrived')) return 'new';
  return 'unknown';
}
function escapeHtml(text) {
  return (text || '').toString().replace(/[&<>'"]/g, function(c) {
    return {'&':'&amp;','<':'&lt;','>':'&gt;',"'":'&#39;','"':'&quot;'}[c];
  });
}
function populateCategories() {
  const cats = [...new Set(BOOKS.map(b => b.category).filter(Boolean))].sort((a,b)=>a.localeCompare(b));
  cats.forEach(cat => {
    const opt = document.createElement('option');
    opt.value = cat; opt.textContent = cat;
    categoryEl.appendChild(opt);
  });
}
function filteredBooks() {
  const q = normalise(searchEl.value);
  const cat = categoryEl.value;
  const st = statusEl.value;
  let rows = BOOKS.filter(b => {
    const hay = normalise([b.title,b.author,b.category,b.subtype,b.status].join(' '));
    return (!q || hay.includes(q)) && (cat === 'all' || b.category === cat) && (st === 'all' || b.status === st);
  });
  const sort = sortEl.value;
  rows.sort((a,b) => normalise(a[sort] || '').localeCompare(normalise(b[sort] || '')) || normalise(a.title).localeCompare(normalise(b.title)));
  return rows;
}
function updateStats(rows) {
  document.getElementById('totalCount').textContent = rows.length;
  document.getElementById('availableCount').textContent = rows.filter(b => statusClass(b.status)==='available').length;
  document.getElementById('inUseCount').textContent = rows.filter(b => statusClass(b.status)==='inuse').length;
  document.getElementById('newCount').textContent = rows.filter(b => statusClass(b.status)==='new').length;
}
function render() {
  visibleBooks = filteredBooks();
  updateStats(visibleBooks);
  emptyEl.style.display = visibleBooks.length ? 'none' : 'block';
  grid.innerHTML = visibleBooks.map(book => {
    const cls = statusClass(book.status);
    const rtl = book.rtl ? ' rtl' : '';
    return `<article class="book" data-id="${book.id}">
      <div class="cover" data-cover-id="${book.id}">
        <span class="cover-status">Cover lookup</span>
        <div class="placeholder">${escapeHtml(book.title).slice(0,60)}</div>
      </div>
      <div class="body${rtl}">
        <div class="title">${escapeHtml(book.title)}</div>
        <div class="author">${book.author ? 'by ' + escapeHtml(book.author) : 'Author not listed'}</div>
        <div class="meta">
          <span class="pill">${escapeHtml(book.category)}</span>
          ${book.subtype ? `<span class="pill">${escapeHtml(book.subtype)}</span>` : ''}
        </div>
        <div class="status ${cls}">${escapeHtml(book.status)}</div>
      </div>
    </article>`;
  }).join('');
  setupObserver();
}
function setupObserver() {
  if (observer) observer.disconnect();
  observer = new IntersectionObserver(entries => {
    entries.forEach(entry => {
      if (entry.isIntersecting) {
        const cover = entry.target;
        observer.unobserve(cover);
        const book = visibleBooks.find(b => b.id === cover.dataset.coverId);
        if (book) loadCover(book, cover);
      }
    });
  }, {rootMargin:'250px'});
  document.querySelectorAll('[data-cover-id]').forEach(el => observer.observe(el));
}
async function loadCover(book, coverEl) {
  if (book.manualImage) { return setCover(coverEl, book.manualImage, 'Manual cover'); }
  const key = coverKey(book);
  if (coverCache[key] === null) { return markNoCover(coverEl); }
  if (coverCache[key]) { return setCover(coverEl, coverCache[key], 'Google Books cover'); }
  const titleQ = book.title.replace(/\([^)]*\)/g,'').trim();
  const q = encodeURIComponent('intitle:' + titleQ + (book.author ? ' inauthor:' + book.author : ''));
  const url = 'https://www.googleapis.com/books/v1/volumes?q=' + q + '&maxResults=1&printType=books&projection=lite';
  try {
    const res = await fetch(url);
    if (!res.ok) throw new Error('cover lookup failed');
    const data = await res.json();
    let img = data.items && data.items[0] && data.items[0].volumeInfo && data.items[0].volumeInfo.imageLinks && (data.items[0].volumeInfo.imageLinks.thumbnail || data.items[0].volumeInfo.imageLinks.smallThumbnail);
    if (img) {
      img = img.replace('http://','https://').replace('zoom=1','zoom=2');
      coverCache[key] = img; saveCoverCache();
      setCover(coverEl, img, 'Google Books cover');
    } else {
      coverCache[key] = null; saveCoverCache();
      markNoCover(coverEl);
    }
  } catch(e) {
    markNoCover(coverEl, 'Cover unavailable');
  }
}
function setCover(coverEl, src, label) {
  const safeLabel = escapeHtml(label || 'Cover');
  coverEl.innerHTML = `<span class="cover-status">${safeLabel}</span><img src="${escapeHtml(src)}" alt="Book cover" loading="lazy" onerror="this.parentElement.innerHTML='<span class=&quot;cover-status&quot;>No cover found</span><div class=&quot;placeholder&quot;>Cover coming soon</div>'">`;
}
function markNoCover(coverEl, label='No cover found') {
  const existing = coverEl.querySelector('.placeholder');
  const text = existing ? existing.textContent : 'Cover coming soon';
  coverEl.innerHTML = `<span class="cover-status">${escapeHtml(label)}</span><div class="placeholder">${escapeHtml(text)}</div>`;
}
